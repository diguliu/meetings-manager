  def disjoin
        @participant = Participant.find(params[:participant][:id])
        @meeting = Meeting.find(params[:meeting][:id])

        respond_to do |format|
                if @participant.meetings.delete(@meeting)
                        flash[:notice] = "Participant left the meeting."
                        format.html {redirect_to(@participant)}
                        format.xml {head :ok}
                else
                        flash[:notice] = "Failed to left the meeting."
                        format.html {redirect_to(@participant)}
                format.xml  { render :xml => @participant.errors, :status => :unprocessable_entity }
                end
        end
  end
