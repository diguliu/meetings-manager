  def test_iso_format
    { '2006-01-01' => '2006-01-01', '1900-04-22' => '1900-04-22',
      '2008-03-04T20:33:41' => '2008-03-04' }.each do |value, result|
      assert_update_and_equal result, :date_of_birth => value
    end
  end
